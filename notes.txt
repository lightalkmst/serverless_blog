Table: User

Table: Posts
  id PRIMARY KEY INTEGER,
  user_id FOREIGN KEY INTEGER,
  title TEXT,
  post TEXT,

  timestamp TIMESTAMPTZ,
  tags TEXT,

Table: Comments
  id INTEGER,
  user_id FOREIGN KEY INTEGER,
  replied_comment_id INTEGER,
  comment TEXT,
  timestamp TIMESTAMPTZ,

Table: Messages
  id INTEGER,
  user_from_id FOREIGN KEY INTEGER,
  user_to_id FOREIGN KEY INTEGER,
  message TEXT,
  timestamp TIMESTAMPTZ,
